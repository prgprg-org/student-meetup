\def\frame#1{%
    \hbox{%
        \vrule
	\vtop{%
	    \vbox{%
	        \hrule
                \kern1pt
                \hbox{%
                    \kern1pt
                    {#1}%
                    \kern1pt
                }%
            }%
            \kern1pt
            \hrule
        }%
        \vrule
    }%
}

\def\shipout{\directlua{
    local function cb(head)
        require("nodetree").print(head)
        require("viznodelist").nodelist_visualize(head, "\jobname.gv")
        return head
    end
    local head = token.scan_list()
    tex.box[0] = head
    cb(head)
    tex.shipout(0)
}}
